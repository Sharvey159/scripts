local HttpService = game:GetService("HttpService")
local Workspace = game:GetService("Workspace")

local request = request or http_request or (syn and syn.request)
assert(request, "HTTP requests are not supported by this executor.")

local WEBHOOK_URL = "https://discord.com/api/webhooks/1521577385122730205/4SCZjVoN91ZuCHDoQiZxTiA20X3lIefGWuPJ4Q1aQmY8OxAHs5etNv1LzcEapytl5H8w"

local lastPhase = nil
local lastWeather = nil

local function sendWebhook(title, body, color, mention)
	task.spawn(function()
		local success, err = pcall(function()
			request({
				Url = WEBHOOK_URL,
				Method = "POST",
				Headers = {
					["Content-Type"] = "application/json"
				},
				Body = HttpService:JSONEncode({
					content = mention or "",
					allowed_mentions = {
						users = { "604609802835853333" },
						roles = { "1521573894383861851" }
					},
					embeds = {
						{
							title = title,
							description = body,
							color = color or 3447003,
							timestamp = os.date("!%Y-%m-%dT%H:%M:%SZ")
						}
					}
				})
			})
		end)

		if not success then
			warn("[PhaseWatcher] Webhook failed:", err)
		end
	end)
end

-- Startup notification
sendWebhook(
	"👀 Phase Watcher",
	"✅ Phase Watcher is now online and monitoring Phase & Weather changes.",
	0x2ECC71
)

local function checkTimeCycle()
	local phase = Workspace:GetAttribute("ActivePhase")
	local weather = Workspace:GetAttribute("ActiveWeather")

	-- Wait until both attributes exist
	if not phase or not weather then
		return
	end

	-- Prevent duplicate notifications
	if phase == lastPhase and weather == lastWeather then
		return
	end

	lastPhase = phase
	lastWeather = weather

	local title = "🌗 Phase Changed"
	local color = 0xF1C40F
	local mention = nil
	local icon = "🌗"

	if phase == "Day" then
		icon = "☀️"
		title = "☀️ Day Time"
		color = 0xF1C40F

	elseif phase == "Sunset" then
		icon = "🌇"
		title = "🌇 Sunset"
		color = 0xE67E22

	elseif phase == "Night" then
		icon = "🌙"

		if weather == "Bloodmoon" then
			title = "🩸🌕 BLOOD MOON HAS BEGUN!"
			color = 0xC70039
			mention = "<@&1521573894383861851>"

		elseif weather == "Mega Moon" then
			title = "🔵🌕 MEGA MOON HAS BEGUN!"
			color = 0x3498DB

		elseif weather == "Rainbow Moon" then
			title = "🌈🌕 RAINBOW MOON HAS BEGUN!"
			color = 0x9B59B6

		elseif weather == "Goldmoon" then
			title = "🟡🌕 GOLD MOON HAS BEGUN!"
			color = 0xF1C40F

		else
			title = "🌙 Night Time Begins"
			color = 0x2C3E50
		end
	end

	local body = string.format(
		"**Phase:** %s %s\n**Type:** %s",
		icon,
		phase,
		weather
	)

	sendWebhook(title, body, color, mention)
end

-- Listen for changes
Workspace:GetAttributeChangedSignal("ActivePhase"):Connect(checkTimeCycle)
Workspace:GetAttributeChangedSignal("ActiveWeather"):Connect(checkTimeCycle)

-- Wait for the game to initialize attributes
task.spawn(function()
	while Workspace:GetAttribute("ActivePhase") == nil
		or Workspace:GetAttribute("ActiveWeather") == nil do
		task.wait(1)
	end

	checkTimeCycle()
end)
