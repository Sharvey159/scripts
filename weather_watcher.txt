local ReplicatedStorage = game:GetService("ReplicatedStorage")
local HttpService = game:GetService("HttpService")
local WeatherValues = ReplicatedStorage:WaitForChild("WeatherValues")
local request = request or http_request or (syn and syn.request)
assert(request, "HTTP requests are not supported by this executor.")
local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer

-- Replaced AVOID_WEATHER with SAFE_WEATHER. 
-- Any weather NOT in this list will trigger a kick.
local SAFE_WEATHER = {
    Rain = true,
    Sunburst = true,
}

local WEBHOOK_URL =
	"https://discord.com/api/webhooks/1520090710290731101/wBfz1Eg2q2nvSNMBYKeu0ivklrMD-7DtKpLa5zHShJbj6-Js2cfBIP_ynFrQkd9ZprYD"
local connections = {}
local updating = {}

local function sendWebhook(title, body, color, mention)
	task.spawn(function()
		local success, err = pcall(function()
			request({
				Url = WEBHOOK_URL,
				Method = "POST",
				Headers = { ["Content-Type"] = "application/json" },
				Body = HttpService:JSONEncode({
					content = mention or "",
					-- Updated to allow dynamic parsing of both users and roles passed in the "content"
					allowed_mentions = { parse = { "users", "roles" } },
					embeds = {
						{
							title = title,
							description = body,
							color = color or 3447003,
							timestamp = os.date("!%Y-%m-%dT%H:%M:%SZ"),
						},
					},
				}),
			})
		end)
		if not success then
			warn("[Weather Watcher] Webhook failed:", err)
		end
	end)
end

sendWebhook("👀 Weather Watcher", "✅ Weather Watcher is now online and monitoring weather events.", 0x2ECC71)

local function formatTime(seconds)
	seconds = math.max(0, math.floor(seconds))

	local minutes = math.floor(seconds / 60)
	local secs = seconds % 60

	if minutes > 0 then
		return string.format("%dm %02ds", minutes, secs)
	end

	return string.format("%ds", secs)
end

local function sendWeather(weatherName)
    task.spawn(function()
        local endTime = WeatherValues:GetAttribute(weatherName .. "_EndTime") or 0

        if endTime <= 0 then
            updating[weatherName] = nil
            return
        end
        
        local remaining = math.max(0, endTime - os.time())
        local formattedRemaining = formatTime(remaining)

        -- If weather is NOT Rain or Sunburst, initiate leave protocol
        if not SAFE_WEATHER[weatherName] then
            local body = string.format(
                "**Weather:** %s\n" ..
                "🚪 Action: Leaving server\n" ..
                "⏱ Rejoin after: %s\n" ..
                "📅 Ends: <t:%d:R>\n" ..
                "🕒 Exact: <t:%d:F>",
                weatherName,
                formattedRemaining,
                endTime,
                endTime
            )

            -- Ping you so you know to rejoin
            sendWebhook("🚨 Avoided Weather", body, 0xE74C3C, "<@604609802835853333>")

            task.wait(1) -- Give the webhook time to send.

            -- Added the duration to the kick message
            LocalPlayer:Kick(
                string.format(
                    "Avoiding %s.\nRejoin in %s.",
                    weatherName,
                    formattedRemaining
                )
            )

            return
        end

        -- Safe Weather Webhook Logic
        local body = string.format(
            "**Weather:** %s\n⏳ Ends: <t:%d:R>\n📅 Exact: <t:%d:F>",
            weatherName,
            endTime,
            endTime
        )
        
        local mentionStr = "<@604609802835853333>"
        
        -- Add the specific role ping only when Sunburst appears
        if weatherName == "Sunburst" then
            mentionStr = mentionStr .. " <@&1522844669027815426>"
        end

        sendWebhook("🌦 Weather Event", body, 0x00BFFF, mentionStr)

        updating[weatherName] = nil
    end)
end

local function watchWeather(weatherName)
	local playingAttribute = weatherName .. "_Playing"

	if connections[playingAttribute] then
		return
	end

	connections[playingAttribute] = WeatherValues:GetAttributeChangedSignal(playingAttribute):Connect(function()
		if WeatherValues:GetAttribute(playingAttribute) and not updating[weatherName] then
			updating[weatherName] = true
			sendWeather(weatherName)
		end
	end)

	-- If already active when the script starts
	if WeatherValues:GetAttribute(playingAttribute) and not updating[weatherName] then
		updating[weatherName] = true
		sendWeather(weatherName)
	end
end

local function scanAttributes()
	for attributeName in pairs(WeatherValues:GetAttributes()) do
		local weatherName = attributeName:match("^(.*)_Playing$")

		if weatherName and typeof(WeatherValues:GetAttribute(attributeName)) == "boolean" then
			watchWeather(weatherName)
		end
	end
end

-- Initial scan
scanAttributes()

-- Keep watching for newly-added weather attributes
while true do
	task.wait(5)
	scanAttributes()
end
