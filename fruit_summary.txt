---------------------------------------------------
-- Dragging
---------------------------------------------------

local dragging = false
local dragInput
local dragStart
local startPos

titleBar.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 then
		dragging = true
		dragStart = input.Position
		startPos = frame.Position

		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

titleBar.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement then
		dragInput = input
	end
end)

UIS.InputChanged:Connect(function(input)
	if input == dragInput and dragging then
		local delta = input.Position - dragStart
		frame.Position =
			UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
	end
end)

local WEBHOOK_URL =
	"https://discord.com/api/webhooks/1520269449251262494/kIdsK1mVMEVqvZblghzbB7NSIMF4KF9NB13phNN0eAqYFsoeymHaYIkWEoJFKTOiVjBj"

local function sendWebhook(title, body, color)
	task.spawn(function()
		local success, err = pcall(function()
			local response = request({
				Url = WEBHOOK_URL,
				Method = "POST",
				Headers = {
					["Content-Type"] = "application/json",
				},
				Body = HttpService:JSONEncode({
					username = "Fruit Logger",
					embeds = {
						{
							title = title,
							description = body,
							color = color or 3447003,
							timestamp = DateTime.now():ToIsoDate(),
						},
					},
				}),
			})

			if response and response.StatusCode ~= 204 and response.StatusCode ~= 200 then
				warn("Webhook failed:", response.StatusCode, response.Body)
			end
		end)

		if not success then
			warn("Webhook Error:", err)
		end
	end)
end

local function buildWebhookText()
	local lines = {}

	local lines = {}

	table.insert(lines, "# Fruit Summary")
	table.insert(lines, "")

	-- Total Fruits
	table.insert(lines, ("Total Fruits: **%d**"):format(totalFruitCount))
	table.insert(lines, "")

	-- Mutation Summary
	table.insert(lines, "## Mutation Summary")
	for mutation, count in pairs(mutationCounts) do
		table.insert(lines, ("• %s: %d"):format(mutation, count))
	end

	table.insert(lines, "")

	-- Weight Ranges
	table.insert(lines, "## Weight Ranges")
	for _, range in ipairs(weightRanges) do
		table.insert(lines, ("%s kg : %d"):format(range.label, range.count))
	end

	table.insert(lines, "")

	-- Sorted Fruits
	table.insert(lines, "## Fruits (Highest KG)")
	for _, fruit in ipairs(sortedFruits) do
		table.insert(lines, ("%s - %.2f kg - %s"):format(fruit.name, fruit.kg, fruit.mutation))
	end

	-- This is what gets sent to Discord
	webhookText = table.concat(lines, "\n")

	return table.concat(lines, "\n")
end
