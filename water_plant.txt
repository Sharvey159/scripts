local Players = game:GetService("Players")

local player = Players.LocalPlayer
local playerGui = player:WaitForChild("PlayerGui")

local WATERING_CAN_NAME = "Super Watering Can"

local function getCharacter()
	return player.Character or player.CharacterAdded:Wait()
end

local function equipWateringCan()
	local character = getCharacter()

	local tool =
		player.Backpack:FindFirstChild(WATERING_CAN_NAME)
		or character:FindFirstChild(WATERING_CAN_NAME)

	if not tool then
		warn(WATERING_CAN_NAME .. " not found.")
		return
	end

	if tool.Parent ~= character then
		local humanoid = character:FindFirstChild("Humanoid")
		if humanoid then
			humanoid:EquipTool(tool)
		end
	end
end

local function createUI()
	local existing = playerGui:FindFirstChild("WaterAdminPanel")
	if existing then
		existing:Destroy()
	end

	local gui = Instance.new("ScreenGui")
	gui.Name = "WaterAdminPanel"
	gui.ResetOnSpawn = false
	gui.Parent = playerGui

	local frame = Instance.new("Frame")
	frame.Size = UDim2.new(0, 220, 0, 70)
	frame.Position = UDim2.new(0.5, -110, 0.8, -70)
	frame.BackgroundColor3 = Color3.fromRGB(25, 25, 30)
	frame.BorderSizePixel = 0
	frame.Parent = gui

	Instance.new("UICorner", frame).CornerRadius = UDim.new(0, 8)

	local layout = Instance.new("UIListLayout")
	layout.HorizontalAlignment = Enum.HorizontalAlignment.Center
	layout.VerticalAlignment = Enum.VerticalAlignment.Center
	layout.Parent = frame

	local button = Instance.new("TextButton")
	button.Size = UDim2.new(0.9, 0, 0, 35)
	button.BackgroundColor3 = Color3.fromRGB(80, 170, 255)
	button.Text = "Equip Watering Can"
	button.TextColor3 = Color3.new(1, 1, 1)
	button.Font = Enum.Font.GothamBold
	button.TextSize = 14
	button.Parent = frame

	Instance.new("UICorner", button).CornerRadius = UDim.new(0, 4)

	button.MouseButton1Click:Connect(function()
		equipWateringCan()
	end)
end

createUI()
