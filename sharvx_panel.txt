local TweenConfig = TweenInfo.new(0.25, Enum.EasingStyle.Quart, Enum.EasingDirection.Out)

-- Helper: Dragging Logic
local function EnableDragging(handle, target)
    local dragging, dragInput, dragStart, startPos
    
    handle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = target.Position
            
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)
    
    handle.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)
    
    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            local delta = input.Position - dragStart
            target.Position = UDim2.new(
                startPos.X.Scale, startPos.X.Offset + delta.X,
                startPos.Y.Scale, startPos.Y.Offset + delta.Y
            )
        end
    end)
end

-- Helper: Hover Animations
local function ApplyHover(button)
    button.MouseEnter:Connect(function()
        TweenService:Create(button, TweenConfig, {BackgroundColor3 = Theme.Hover}):Play()
    end)
    button.MouseLeave:Connect(function()
        TweenService:Create(button, TweenConfig, {BackgroundColor3 = Theme.Panels}):Play()
    end)
end

-- Component: Script Button

    Button.BackgroundColor3 = Theme.Panels
    Button.Text = scriptInfo.Name
    Button.TextColor3 = Theme.Text

    Button.TextSize = 13
    Button.AutoButtonColor = false
    Button.Parent = parent
    
    local Corner = Instance.new("UICorner")
    Corner.CornerRadius = UDim.new(0, 6)

-- Module Entry Point
function UI.Create(callbacks, registry)
    local UI_NAME = "SharveyScriptManagerUI"
    if CoreGui:FindFirstChild(UI_NAME) then
        CoreGui[UI_NAME]:Destroy()
    end

    local ScreenGui = Instance.new("ScreenGui")
    ScreenGui.Name = UI_NAME
    ScreenGui.ResetOnSpawn = false
    ScreenGui.Parent = CoreGui
    
    -- Main Wrapper handles the footprint and layout
    local Wrapper = Instance.new("Frame")
    Wrapper.Size = UDim2.new(0, 280, 0, 400)
    Wrapper.Position = UDim2.new(0.5, -140, 0.5, -200)
    Wrapper.BackgroundTransparency = 1
    Wrapper.Parent = ScreenGui
    
    local MainFrame = Instance.new("Frame")
    MainFrame.Size = UDim2.new(1, 0, 1, 0)
    MainFrame.BackgroundColor3 = Theme.Background
    MainFrame.ClipsDescendants = true
    MainFrame.Parent = Wrapper
    
    local MainCorner = Instance.new("UICorner")
    MainCorner.CornerRadius = UDim.new(0, 8)
    MainCorner.Parent = MainFrame
    
    -- Generate Layout
    local TopBar, MinBtn, CloseBtn, TitleLbl = CreateTopBar(MainFrame)
    local ScrollList = CreateScrollList(MainFrame)
    local MiniLauncher, RestoreBtn = CreateMiniLauncher(ScreenGui)
    
    -- Dynamically generate script buttons
    for _, scriptInfo in ipairs(registry) do
        CreateScriptButton(scriptInfo, ScrollList, callbacks)
    end
    
    -- Connect dragging (Main Window dragged by TopBar, Launcher dragged by itself)
    EnableDragging(TopBar, Wrapper)
    EnableDragging(MiniLauncher, MiniLauncher)
    
    -- API Object
    local windowAPI = {}
    local minimized = false
    
    function windowAPI:Minimize()
        if minimized then return end
        minimized = true
        
        -- Sync position to ensure the launcher appears right where the window was
        MiniLauncher.Position = Wrapper.Position
        
        Wrapper.Visible = false
        MiniLauncher.Size = UDim2.new(0, 40, 0, 35) -- Start small
        MiniLauncher.Visible = true
        
        TweenService:Create(MiniLauncher, TweenConfig, {Size = UDim2.new(0, 130, 0, 35)}):Play()
    end
    
    function windowAPI:Restore()
        if not minimized then return end
        minimized = false
        
        -- Sync main window back to where the launcher was dragged
        Wrapper.Position = MiniLauncher.Position
        
        MiniLauncher.Visible = false
        Wrapper.Visible = true
    end
    
    function windowAPI:Destroy()
        ScreenGui:Destroy()
    end
    
    function windowAPI:SetTitle(text)
        TitleLbl.Text = text
    end
    
    -- Wire internal UI events
    MinBtn.MouseButton1Click:Connect(function() windowAPI:Minimize() end)
    RestoreBtn.MouseButton1Click:Connect(function() windowAPI:Restore() end)
    
    CloseBtn.MouseButton1Click:Connect(function()
        if callbacks.Close then
            callbacks.Close()
        end
    end)
    
    return windowAPI
end

local function LoadScript(scriptInfo)
    local url = scriptInfo.File
    
    if not string.match(url, "^https?://") then
        url = BASE_URL .. url
    end
    
    local finalUrl = url
    if string.find(finalUrl, "?") then
        finalUrl = finalUrl .. "&t=" .. os.time()
    else
        finalUrl = finalUrl .. "?t=" .. os.time()
    end
    
    local success, err = pcall(function()
        loadstring(game:HttpGet(finalUrl))()
    end)
    
    if success then
        print("Loaded:\n" .. scriptInfo.Name)
    else
        warn("Failed:\n" .. scriptInfo.Name .. "\n" .. tostring(err))
    end
end

local function ReloadAllScripts()
    for _, scriptInfo in ipairs(ScriptRegistry) do
        LoadScript(scriptInfo)
    end
end

local function AutoLoadScripts()
    for _, scriptInfo in ipairs(ScriptRegistry) do
        local configKey = string.gsub(scriptInfo.Name, "%s+", "")
        if Config.AutoLoad[configKey] then
            LoadScript(scriptInfo)
        end
    end
end
