local function createQueue()
    return {
        items = {},
        first = 1,
        last = 0,
        push = function(self, item)
            self.last += 1
            self.items[self.last] = item
        end,
        pop = function(self)
            if self.first > self.last then return nil end
            local val = self.items[self.first]
            self.items[self.first] = nil
            self.first += 1
            if self.first > self.last then 
                self.first = 1
                self.last = 0
            end
            return val
        end,
        count = function(self)
            return self.last - self.first + 1
        end
    }
end

local function sendWebhook(title, body, color)
    task.spawn(function()
        pcall(function()
            request({
                Url = WEBHOOK_URL,
                Method = "POST",
                Headers = { ["Content-Type"] = "application/json" },
                Body = HttpService:JSONEncode({
                    embeds = {{
                        title = title,
                        description = body,
                        color = color or 3447003,
                        timestamp = os.date("!%Y-%m-%dT%H:%M:%SZ"),
                    }},
                }),
            })
        end)
    end)
end

local function getCharacter()
    return player.Character or player.CharacterAdded:Wait()
end

local cachedShovelInstance = nil
local function getShovelTool()
    local character = player.Character
    if not character then return nil end
    
    if cachedShovelInstance and cachedShovelInstance.Parent == character then
        return cachedShovelInstance
    end
    
    local tool = character:FindFirstChild(SHOVEL_NAME)
    if tool then
        cachedShovelInstance = tool
        return tool
    end
    
    tool = player.Backpack:FindFirstChild(SHOVEL_NAME)
    if tool then
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        if humanoid then
            humanoid:EquipTool(tool)
            cachedShovelInstance = tool
            return tool
        end
    end
    
    return nil
end

local function equipTool(toolName)
    local char = getCharacter()
    local tool = player.Backpack:FindFirstChild(toolName) or char:FindFirstChild(toolName)
    if not tool then return nil end
    
    if tool.Parent ~= char then
        local humanoid = char:FindFirstChild("Humanoid")
        if humanoid then 
            humanoid:EquipTool(tool) 
        end
        
        local deadline = tick() + 2
        repeat task.wait() until tool.Parent == char or tick() > deadline
        task.wait(0.1)
    end
    return tool
end

local function getPlantPosition(plant)
    if not plant then return nil end
    if plant:IsA("Model") and plant.PrimaryPart then return plant.PrimaryPart.Position end
    
    local ok, pivot = pcall(function() return plant:GetPivot() end)
    if ok and pivot then return pivot.Position end
    if plant:IsA("BasePart") then return plant.Position end
    
    for _, child in ipairs(plant:GetDescendants()) do
        if child:IsA("BasePart") then return child.Position end
    end
    return nil
end

local function getFirstAvailablePlant()
    local plantsFolder = myPlot:FindFirstChild("Plants")
    if not plantsFolder then return nil end
    
    local plants = plantsFolder:GetChildren()
    for _, plant in ipairs(plants) do
        if getPlantPosition(plant) then return plant end
    end
    return plants[1]
end

local function getPlayerGroundHeight()
    local char = getCharacter()
    local root = char:FindFirstChild("HumanoidRootPart") or char.PrimaryPart
    if not root then return 0 end
    
    local params = RaycastParams.new()
    params.FilterType = Enum.RaycastFilterType.Exclude
    params.FilterDescendantsInstances = { char }
    
    local origin = root.Position
    local direction = Vector3.new(0, -100, 0)
    local result = workspace:Raycast(origin, direction, params)
    
    return result and result.Position.Y or origin.Y
end

local function placeSprinklerAt(position, sprinklerName, plotId)
    local tool = equipTool(sprinklerName)
    if not tool then return false end
    
    pcall(function()
        Networking.Place.PlaceSprinkler:Fire(position, sprinklerName, tool, plotId)
    end)
    return true
end

local function runMaintenanceCycle()
    local sprinklersFolder = myPlot:FindFirstChild("Sprinklers")
    if not sprinklersFolder then return end

    local targetPlant = getFirstAvailablePlant()
    local plantPos = targetPlant and getPlantPosition(targetPlant)

    if not plantPos then return end
    
    local groundY = getPlayerGroundHeight()
    local centerPosition = Vector3.new(plantPos.X, groundY, plantPos.Z)

    local allVerified = false

    -- Verification Loop
    while not allVerified and isScriptRunning do
        local activeSprinklers = {}
        
        for _, sprinkler in ipairs(sprinklersFolder:GetChildren()) do
            local sName = sprinkler:GetAttribute("SprinklerName") or sprinkler.Name
            if sName then
                activeSprinklers[sName] = true
            end
        end

        if activeSprinklers[SUPER_SPRINKLER_NAME] then
            allVerified = true
            break
        end

        local hasAllStandard = true
        for _, sType in ipairs(SPRINKLER_TYPES) do
            if not activeSprinklers[sType] then
                hasAllStandard = false
                break
            end
        end

        if hasAllStandard then
            allVerified = true
            break
        end

        local missingSprinklers = {}
        local char = getCharacter()
        local ownsSuperSprinkler = player.Backpack:FindFirstChild(SUPER_SPRINKLER_NAME) or (char and char:FindFirstChild(SUPER_SPRINKLER_NAME))

        if ownsSuperSprinkler then
            table.insert(missingSprinklers, SUPER_SPRINKLER_NAME)
        else
            for _, sType in ipairs(SPRINKLER_TYPES) do
                if not activeSprinklers[sType] then
                    table.insert(missingSprinklers, sType)
                end
            end
        end

        local toolsSuccessfullyPlaced = 0

        for k, missingType in ipairs(missingSprinklers) do
            local angle = (k - 1) * (2 * math.pi / 8)
            local offset = Vector3.new(math.cos(angle) * DEFAULT_RADIUS, 0, math.sin(angle) * DEFAULT_RADIUS)
            
            local success = placeSprinklerAt(centerPosition + offset, missingType, myPlotId)
            if success then
                toolsSuccessfullyPlaced += 1
                task.wait(DEFAULT_DELAY)
            end
        end
        
        task.wait(2)

        if toolsSuccessfullyPlaced == 0 and not allVerified then
            warn("Sprinklers are missing, but no valid tools were found to place. Exiting maintenance to prevent softlock.")
            break 
        end
    end

    if allVerified then
        local waterTool = equipTool(WATERING_CAN_NAME)
        if waterTool then
            pcall(function()
                Networking.WateringCan.UseWateringCan:Fire(centerPosition, WATERING_CAN_NAME, waterTool)
            end)
            task.wait(0.5) 
        end
    end
end

local function removeKeptFruitState(data)
    if not data.FruitId then return end
    if keptFruitCache[data.FruitId] then
        keptFruitCache[data.FruitId] = nil
        keptCount -= 1
        
        for i, f in ipairs(keptFruitList) do
            if f.FruitId == data.FruitId then
                table.remove(keptFruitList, i)
                break
            end
        end
    end
end

local function processFruit(fruit, plant)
    if not fruit or fruitCache[fruit] then return end
    
    local name = fruit:GetAttribute("CorePartName")
    local id = fruit:GetAttribute("FruitId")
    if not id or not name then return end
    
    local plantId = plant:GetAttribute("PlantId") or plant.Name
    if not plantId then return end
    
    local mutation = fruit:GetAttribute("Mutation")
    if not mutation or mutation == "" then
        mutation = "None"
    end
    
    local weight = FruitVisualizerController:CalculateFruitWeight(fruit)
    if not weight and FruitVisualizerController.CalculatePlantWeight then
        weight = FruitVisualizerController:CalculatePlantWeight(fruit)
    end
    weight = tonumber(weight) or 0
    
    local fruitData = {
        PlantId = plantId,
        FruitId = id,
        Name = name,
        Weight = weight,
        Mutation = mutation,
        Instance = fruit,
        Processed = false
    }
    
    fruitCache[fruit] = fruitData
    
    if weight < 100 then
        -- Calculate the expected time each worker will take to clear its queue
        local expectedHarvestTime = harvestQueue:count() * HARVEST_INTERVAL
        local expectedShovelTime = shovelQueue:count() * SHOVEL_INTERVAL
        
        -- Delegate the task to whichever queue can finish it sooner
        if expectedShovelTime < expectedHarvestTime then
            shovelQueue:push(fruitData)
        else
            harvestQueue:push(fruitData)
        end
    else
        -- Kept fruits (100kg+)
        if not keptFruitCache[id] then
            keptFruitCache[id] = true
            keptCount += 1
            
            table.insert(keptFruitList, fruitData)
            table.sort(keptFruitList, function(a, b) return a.Weight > b.Weight end)
            webhookQueue:push(fruitData)
        end
    end
end

local function setupPlant(plant)
    if plantConnections[plant] then return end
    plantConnections[plant] = {}
    
    local fruitsFolder = plant:WaitForChild("Fruits", 5)
    if not fruitsFolder then return end
    
    for _, fruit in ipairs(fruitsFolder:GetChildren()) do processFruit(fruit, plant) end
    
    local addedConn = fruitsFolder.ChildAdded:Connect(function(fruit) task.defer(processFruit, fruit, plant) end)
    local removedConn = fruitsFolder.ChildRemoved:Connect(function(fruit)
        local data = fruitCache[fruit]
        if data then
            data.Processed = true 
            fruitCache[fruit] = nil
            if data.Weight >= 100 then removeKeptFruitState(data) end
        end
    end)
    
    table.insert(plantConnections[plant], addedConn)
    table.insert(plantConnections[plant], removedConn)
end

local function cleanupPlant(plant)
    local conns = plantConnections[plant]
    if conns then
        for _, conn in ipairs(conns) do conn:Disconnect() end
        plantConnections[plant] = nil
    end
    
    local fruitsFolder = plant:FindFirstChild("Fruits")
    if fruitsFolder then
        for _, fruit in ipairs(fruitsFolder:GetChildren()) do
            local data = fruitCache[fruit]
            if data then
                data.Processed = true
                fruitCache[fruit] = nil
                if data.Weight >= 100 then removeKeptFruitState(data) end
            end
        end
    end
end

for _, plant in ipairs(myPlot.Plants:GetChildren()) do setupPlant(plant) end
local plantAddedConn = myPlot.Plants.ChildAdded:Connect(setupPlant)
local plantRemovedConn = myPlot.Plants.ChildRemoved:Connect(cleanupPlant)

-- Worker 1: Harvest Engine
task.spawn(function()
    while isScriptRunning do
        if isAutoFarming and not isMaintaining and harvestQueue:count() > 0 then
            local data = harvestQueue:pop()
            if data and not data.Processed and data.Instance and data.Instance.Parent then
                data.Processed = true
                pcall(function() Networking.Garden.CollectFruit:Fire(data.PlantId, data.FruitId or "") end)
                task.wait(HARVEST_INTERVAL)
            end
        else
            task.wait(0.1) 
        end
    end
end)

-- Worker 2: Shovel Engine 
task.spawn(function()
    while isScriptRunning do
        if isAutoFarming and not isMaintaining and shovelQueue:count() > 0 then
            local data = shovelQueue:pop()
            if data and not data.Processed and data.Instance and data.Instance.Parent then
                local tool = getShovelTool()
                if tool then
                    data.Processed = true
                    local shovelType = tool:GetAttribute("Shovel")
                    pcall(function() Networking.Shovel.UseShovel:Fire(data.PlantId, data.FruitId, shovelType, tool) end)
                    task.wait(SHOVEL_INTERVAL)
                else
                    -- Re-queue if tool takes a moment to equip
                    shovelQueue:push(data)
                    task.wait(0.5) 
                end
            end
        else
            task.wait(0.1)
        end
    end
end)

-- Worker 3: Webhook Batching Engine
task.spawn(function()
    while isScriptRunning do
        if webhookQueue:count() > 0 then
            task.wait(2) 
            local batchedFruits = {}
            while webhookQueue:count() > 0 do table.insert(batchedFruits, webhookQueue:pop()) end
            
            if #batchedFruits > 0 then
                local title = string.format("剣 %d New Kept Fruit%s", #batchedFruits, #batchedFruits > 1 and "s" or "")
                local descText = ""
                
                for _, f in ipairs(batchedFruits) do
                    descText ..= string.format("+ %s\n  %.2fkg\n  %s\n\n", f.Name, f.Weight, f.Mutation)
                end
                descText ..= string.format("**Total Kept:** %d", keptCount)
                sendWebhook(title, descText, 16766720) 
                task.wait(0.5)
            end
        else
            task.wait(0.1)
        end
    end
end)

-- Worker 4: Main Coordinator & Maintenance Loop 
task.spawn(function()
    local wasProcessing = false
    while isScriptRunning do
        if isAutoFarming then
            local harvestRemaining = harvestQueue:count()
            local shovelRemaining = shovelQueue:count()
            local totalTasks = harvestRemaining + shovelRemaining 

            if totalTasks > 0 then
                wasProcessing = true
            elseif wasProcessing and totalTasks == 0 then
                -- Harvesting cycle is complete. Lock routines and execute maintenance
                wasProcessing = false
                isMaintaining = true
                
                -- Periodic Processed Fruit Cache Reset
                completedCycles += 1
                if completedCycles >= CYCLE_RESET_THRESHOLD then
                    local cleanedCache = {}
                    for fruit, data in pairs(fruitCache) do
                        if data.Instance and data.Instance.Parent and not data.Processed then
                            cleanedCache[fruit] = data
                        end
                    end
                    fruitCache = cleanedCache
                    completedCycles = 0
                end

                runMaintenanceCycle()
                isMaintaining = false
            end
        end
        task.wait(0.5)
    end
end)
