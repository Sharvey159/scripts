-- High-Performance O(1) Memory-Stable Queue Structure 
local function createQueue()
    return {
        items = {},
        first = 1,
        last = 0,
        push = function(self, item)
            self.last += 1
            self.items[self.last] = item
        end,
        pop = function(self)
            if self.first > self.last then return nil end
            local val = self.items[self.first]
            self.items[self.first] = nil
            self.first += 1
            -- Reset pointers when empty to keep the table array boundary flat and packed
            if self.first > self.last then 
                self.first = 1
                self.last = 0
            end
            return val
        end,
        count = function(self)
            return self.last - self.first + 1
        end
    }
end

-- Webhook Utility Function
local function sendWebhook(title, body, color)
    task.spawn(function()
        pcall(function()
            request({
                Url = WEBHOOK_URL,
                Method = "POST",
                Headers = { ["Content-Type"] = "application/json" },
                Body = HttpService:JSONEncode({
                    embeds = {
                        {
                            title = title,
                            description = body,
                            color = color or 3447003,
                            timestamp = os.date("!%Y-%m-%dT%H:%M:%SZ"),
                        },
                    },
                }),
            })
        end)
    end)
end

-- 4. Shared Utilities & Caching Lookups
local cachedShovelInstance = nil
local function getShovelTool()
    local character = player.Character
    if not character then return nil end
    
    if cachedShovelInstance and cachedShovelInstance.Parent == character then
        return cachedShovelInstance
    end
    
    local tool = character:FindFirstChild(SHOVEL_NAME)
    if tool then
        cachedShovelInstance = tool
        return tool
    end
    
    tool = player.Backpack:FindFirstChild(SHOVEL_NAME)
    if tool then
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        if humanoid then
            humanoid:EquipTool(tool)
            cachedShovelInstance = tool
            return tool
        end
    end
    
    return nil
end

-- 5. Reactive Event System (Producer Core)
local function processFruit(fruit, plant)
    if not fruit or fruitCache[fruit] then return end
    
    local name = fruit:GetAttribute("CorePartName")
    local id = fruit:GetAttribute("FruitId")
    if not id or not name then return end
    
    local plantId = plant:GetAttribute("PlantId") or plant.Name
    if not plantId then return end
    
    local mutation = fruit:GetAttribute("Mutation")
    if not mutation or mutation == "" then
        mutation = "None"
    end
    
    local weight = FruitVisualizerController:CalculateFruitWeight(fruit)
    if not weight and FruitVisualizerController.CalculatePlantWeight then
        weight = FruitVisualizerController:CalculatePlantWeight(fruit)
    end
    weight = tonumber(weight) or 0
    
    local fruitData = {
        PlantId = plantId,
        FruitId = id,
        Name = name,
        Weight = weight,
        Mutation = mutation,
        Instance = fruit,
        Processed = false
    }
    
    fruitCache[fruit] = fruitData
    
    if weight < 50 then
        harvestQueue:push(fruitData)
    elseif weight < 60 then
        shovelQueue:push(fruitData)
    else
        -- Check cache for duplicate protection
        if not keptFruitCache[id] then
            keptFruitCache[id] = true
            keptCount += 1
            
            -- Insert and sort kept array in place
            table.insert(keptFruitList, fruitData)
            table.sort(keptFruitList, function(a, b)
                return a.Weight > b.Weight
            end)
            
            -- Dispatch to webhook worker
            webhookQueue:push(fruitData)
        end
    end
end

-- Cleanup Utility for Kept Fruits
local function removeKeptFruitState(data)
    if not data.FruitId then return end
    if keptFruitCache[data.FruitId] then
        keptFruitCache[data.FruitId] = nil
        keptCount -= 1
        
        -- Clean out from sorted list (O(n) but tiny list)
        for i, f in ipairs(keptFruitList) do
            if f.FruitId == data.FruitId then
                table.remove(keptFruitList, i)
                break
            end
        end
    end
end

local function setupPlant(plant)
    if plantConnections[plant] then return end
    plantConnections[plant] = {}
    
    local fruitsFolder = plant:WaitForChild("Fruits", 5)
    if not fruitsFolder then return end
    
    for _, fruit in ipairs(fruitsFolder:GetChildren()) do
        processFruit(fruit, plant)
    end
    
    local addedConn = fruitsFolder.ChildAdded:Connect(function(fruit)
        task.defer(processFruit, fruit, plant)
    end)
    
    local removedConn = fruitsFolder.ChildRemoved:Connect(function(fruit)
        local data = fruitCache[fruit]
        if data then
            data.Processed = true 
            fruitCache[fruit] = nil
            
            -- Ensure it is removed from active kept pool if it was high weight
            if data.Weight >= 60 then
                removeKeptFruitState(data)
            end
        end
    end)
    
    table.insert(plantConnections[plant], addedConn)
    table.insert(plantConnections[plant], removedConn)
end

local function cleanupPlant(plant)
    local conns = plantConnections[plant]
    if conns then
        for _, conn in ipairs(conns) do
            conn:Disconnect()
        end
        plantConnections[plant] = nil
    end
    
    local fruitsFolder = plant:FindFirstChild("Fruits")
    if fruitsFolder then
        for _, fruit in ipairs(fruitsFolder:GetChildren()) do
            local data = fruitCache[fruit]
            if data then
                data.Processed = true
                fruitCache[fruit] = nil
                
                if data.Weight >= 60 then
                    removeKeptFruitState(data)
                end
            end
        end
    end
end

-- Connect Lifecycle Observers to the Plot
for _, plant in ipairs(myPlot.Plants:GetChildren()) do
    setupPlant(plant)
end

local plantAddedConn = myPlot.Plants.ChildAdded:Connect(setupPlant)
local plantRemovedConn = myPlot.Plants.ChildRemoved:Connect(cleanupPlant)


-- 6. Consumer Threads (FPS-Independent Execution workers)

-- Worker 1: Harvest Engine
task.spawn(function()
    while isScriptRunning do
        if isAutoFarming and harvestQueue:count() > 0 then
            local data = harvestQueue:pop()
            if data and not data.Processed and data.Instance and data.Instance.Parent then
                data.Processed = true
                pcall(function()
                    Networking.Garden.CollectFruit:Fire(data.PlantId, data.FruitId or "")
                end)
                task.wait(HARVEST_INTERVAL)
            end
        else
            task.wait(0.1) 
        end
    end
end)

-- Worker 2: Shovel Engine
task.spawn(function()
    while isScriptRunning do
        if isAutoFarming and shovelQueue:count() > 0 then
            local data = shovelQueue:pop()
            if data and not data.Processed and data.Instance and data.Instance.Parent then
                local tool = getShovelTool()
                if tool then
                    data.Processed = true
                    local shovelType = tool:GetAttribute("Shovel")
                    pcall(function()
                        Networking.Shovel.UseShovel:Fire(data.PlantId, data.FruitId, shovelType, tool)
                    end)
                    task.wait(SHOVEL_INTERVAL)
                else
                    shovelQueue:push(data)
                    task.wait(0.5) 
                end
            end
        else
            task.wait(0.1)
        end
    end
end)

-- Worker 3: Webhook Batching Engine (NEW)
task.spawn(function()
    while isScriptRunning do
        if webhookQueue:count() > 0 then
            -- Initial wait for batched arrivals (e.g., plot load or fast growths)
            task.wait(2) 
            
            local batchedFruits = {}
            while webhookQueue:count() > 0 do
                table.insert(batchedFruits, webhookQueue:pop())
            end
            
            if #batchedFruits > 0 then
                -- Construct summary payload
                local title = string.format("🌕 %d New Kept Fruit%s", #batchedFruits, #batchedFruits > 1 and "s" or "")
                local descText = ""
                
                for _, f in ipairs(batchedFruits) do
                    descText ..= string.format("+ %s\n  %.2fkg\n  %s\n\n", f.Name, f.Weight, f.Mutation)
                end
                
                descText ..= string.format("**Total Kept:** %d", keptCount)
                
                -- Send the webhook to the URL
                sendWebhook(title, descText, 16766720) -- Gold/Yellow embed color
                
                -- Strict cooldown to avoid Discord API drops
                task.wait(0.5)
            end
        else
            task.wait(0.1)
        end
    end
end)

-- Worker 4: UI Throttle & Virtual Text State Caching
task.spawn(function()
    local lastStateText, lastHarvestText, lastShovelText, lastKeptText = "", "", "", ""
    
    while isScriptRunning do
        local harvestRemaining = harvestQueue:count()
        local shovelRemaining = shovelQueue:count()
        
        local harvestStr = "🌿 Harvest State: " .. harvestRemaining
        local shovelStr = "🪓 Shovel State: " .. shovelRemaining
        local keptStr = "🛡️ Kept Plant: " .. keptCount
        local stateStr = "Status: Idle"
        
        if isAutoFarming then
            if harvestRemaining > 0 then
                stateStr = "Status: Harvesting..."
                harvestStr = string.format("🌿 Harvest State: Active (%d left)", harvestRemaining)
            elseif shovelRemaining > 0 then
                stateStr = "Status: Shoveling..."
                shovelStr = string.format("🪓 Shovel State: Active (%d left)", shovelRemaining)
            else
                stateStr = "Status: Processes running in background..."
            end
        else
            stateStr = "Status: Idle (Paused)"
        end
        
        if stateStr ~= lastStateText then lastStateText = stateStr stateLbl.Text = stateStr end
        if harvestStr ~= lastHarvestText then lastHarvestText = harvestStr harvestLbl.Text = harvestStr end
        if shovelStr ~= lastShovelText then lastShovelText = shovelStr shovelLbl.Text = shovelStr end
        if keptStr ~= lastKeptText then lastKeptText = keptStr keptLbl.Text = keptStr end
        
        task.wait(0.3) 
    end
end)
