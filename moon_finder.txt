-- O(1) Queue Implementation (Massive CPU saver for high counts)
local function createQueue()
    return {
        items = {},
        first = 1,
        last = 0,
        push = function(self, item)
            self.last += 1
            self.items[self.last] = item
        end,
        pop = function(self)
            if self.first > self.last then return nil end
            local val = self.items[self.first]
            self.items[self.first] = nil
            self.first += 1
            -- Reset pointers if empty to prevent integer growth
            if self.first > self.last then 
                self.first = 1
                self.last = 0
            end
            return val
        end,
        count = function(self)
            return self.last - self.first + 1
        end
    }
end

-- 4. Shared Utilities
local function equipTool(toolName)
    local character = player.Character
    if not character then return nil end
    local humanoid = character:FindFirstChild("Humanoid")
    if not humanoid then return nil end

    local tool = character:FindFirstChild(toolName) or player.Backpack:FindFirstChild(toolName)
    if not tool then return nil end

    humanoid:EquipTool(tool)
    return tool -- Returning tool directly saves searching for it again
end

-- 5. Thread-Safe Worker Architecture

-- Worker 1: The Scanner (Producer)
task.spawn(function()
    while isScriptRunning do
        if isAutoFarming then
            -- A. Discover New Fruits
            for _, plant in ipairs(myPlot.Plants:GetChildren()) do
                local fruits = plant:FindFirstChild("Fruits")
                if fruits then
                    for _, fruit in ipairs(fruits:GetChildren()) do
                        if not fruitCache[fruit] then
                            fruitCache[fruit] = true
                            
                            local name = fruit:GetAttribute("CorePartName")
                            local id = fruit:GetAttribute("FruitId")
                            if not id or not name then continue end

                            -- Calculate weight ONCE to save CPU
                            local weight = FruitVisualizerController:CalculateFruitWeight(fruit)
                            if not weight and FruitVisualizerController.CalculatePlantWeight then
                                weight = FruitVisualizerController:CalculatePlantWeight(fruit)
                            end
                            weight = tonumber(weight) or 0

                            local fruitData = {
                                PlantId = plant:GetAttribute("PlantId") or plant.Name,
                                FruitId = id,
                                Instance = fruit,
                                Weight = weight
                            }

                            -- Route fruit to appropriate list
                            if weight < 50 then
                                harvestQueue:push(fruitData)
                            elseif weight < 60 then
                                shovelQueue:push(fruitData)
                            else
                                keptCount += 1
                            end
                        end
                    end
                end
            end

            -- B. Cleanup Cache (Prevents Memory Leaks)
            for fruit, _ in pairs(fruitCache) do
                if not fruit.Parent then
                    fruitCache[fruit] = nil
                    processedFruits[fruit] = nil
                    
                    -- If a kept fruit was destroyed by a player, adjust the counter
                    if not processedFruits[fruit] then
                        -- Since we can't easily check weight here, if we wanted perfect tracking
                        -- we'd cache weight too. However, since kept fruits aren't processed by queues,
                        -- a slight counter drift is possible but negligible. For max performance, 
                        -- we clear the cache cheaply.
                    end
                end
            end
        end
        task.wait(LOOP_INTERVAL)
    end
end)

-- Worker 2: The Harvester (Consumer)
task.spawn(function()
    local nextHarvestTime = 0
    while isScriptRunning do
        if isAutoFarming and harvestQueue:count() > 0 then
            local now = os.clock()
            if now >= nextHarvestTime then
                local data = harvestQueue:pop()
                if data and data.Instance.Parent and not processedFruits[data.Instance] then
                    processedFruits[data.Instance] = true
                    
                    pcall(function()
                        Networking.Garden.CollectFruit:Fire(data.PlantId, data.FruitId)
                    end)
                    
                    -- FPS Independent timing
                    nextHarvestTime = now + HARVEST_INTERVAL
                end
            else
                task.wait(math.max(0, nextHarvestTime - now))
            end
        else
            task.wait(0.1) -- Idle sleep
        end
    end
end)

-- Worker 3: The Shoveler (Consumer)
task.spawn(function()
    local nextShovelTime = 0
    while isScriptRunning do
        if isAutoFarming and shovelQueue:count() > 0 then
            local now = os.clock()
            if now >= nextShovelTime then
                local data = shovelQueue:pop()
                if data and data.Instance.Parent and not processedFruits[data.Instance] then
                    
                    local tool = equipTool(SHOVEL_NAME)
                    if tool then
                        processedFruits[data.Instance] = true
                        local shovelType = tool:GetAttribute("Shovel")

                        pcall(function()
                            Networking.Shovel.UseShovel:Fire(
                                data.PlantId,
                                data.FruitId,
                                shovelType,
                                tool
                            )
                        end)
                        
                        -- FPS Independent timing
                        nextShovelTime = now + SHOVEL_INTERVAL
                    else
                        -- Re-queue if shovel is missing so we don't skip the plant
                        shovelQueue:push(data)
                        task.wait(0.5) 
                    end
                end
            else
                task.wait(math.max(0, nextShovelTime - now))
            end
        else
            task.wait(0.1) -- Idle sleep
        end
    end
end)

-- Worker 4: UI Updater (Throttled for Performance)
task.spawn(function()
    local lastUIUpdate = 0
    while isScriptRunning do
        if isAutoFarming then
            local now = os.clock()
            if now - lastUIUpdate > 0.3 then
                lastUIUpdate = now
                
                harvestLbl.Text = "🌿 Harvest Queue: " .. harvestQueue:count()
                shovelLbl.Text = "🪓 Shovel Queue: " .. shovelQueue:count()
                keptLbl.Text = "🛡️ Kept Plant (Lifetime): " .. keptCount

                if harvestQueue:count() > 0 then
                    setState("Harvesting (" .. harvestQueue:count() .. " left)")
                elseif shovelQueue:count() > 0 then
                    setState("Shoveling (" .. shovelQueue:count() .. " left)")
                else
                    setState("Scanning processes active...")
                end
            end
        end
        task.wait(0.1)
    end
end)
